SUBROUTINE INIT_RANDOM_SEED()
	IMPLICIT NONE
	INTEGER :: I, N, CLOCK
	INTEGER, DIMENSION(:), ALLOCATABLE :: SEED

	CALL RANDOM_SEED( SIZE = N )
	ALLOCATE( SEED(1:N) )

	CALL SYSTEM_CLOCK( COUNT = CLOCK )

	SEED = CLOCK + 37 * (/ ( I - 1, I = 1, N ) /)
	CALL RANDOM_SEED( PUT = SEED )

	DEALLOCATE( SEED )
END SUBROUTINE INIT_RANDOM_SEED
